🚀 How to Use These Templates

For Bot Integration:

// 1. Include template CSS <link rel="stylesheet" href="templates.css"> // 2. Use template functions const templates = { chatMessage: (text, isUser) => `<div class="chat-message ${isUser ? 'user' : 'bot'}"> <div class="message-content">${text}</div> <div class="message-time">${new Date().toLocaleTimeString()}</div> </div>`, questionStep: (question, options) => `<div class="question-step"> <h3>${question}</h3> ${options.map(opt => `<button class="option-btn">${opt}</button>`).join('')} </div>`, languageSelector: (languages) => `<div class="language-selector"> ${languages.map(lang => `<div class="lang-option" data-lang="${lang.code}"> <span class="flag">${lang.flag}</span> <span>${lang.name}</span> </div>`).join('')} </div>` }; // 3. Render dynamically document.getElementById('chat-container').innerHTML += templates.chatMessage("Hello! How can I help?", false);

Dynamic Content Generation:

// Generate questionnaire based on website type function generateQuestionnaire(websiteType) { const questions = { basic: [ "What is your business name?", "What pages do you need?", "What is your budget range?" ], dynamic: [ "What features do you need?", "Do you need user login?", "What is your timeline?" ], '3d': [ "What type of 3D experience?", "Do you have 3D models?", "What is your budget?" ] }; return questions[websiteType] || questions.basic; } // Example: Generate language selector const languages = [ {code: 'hi', name: 'हिंदी', flag: '🇮🇳'}, {code: 'en', name: 'English', flag: '🇺🇸'}, {code: 'ta', name: 'தமிழ்', flag: '🇮🇳'}, {code: 'te', name: 'తెలుగు', flag: '🇮🇳'}, {code: 'kn', name: 'ಕನ್ನಡ', flag: '🇮🇳'}, {code: 'ml', name: 'മലയാളം', flag: '🇮🇳'}, {code: 'pa', name: 'ਪੰਜਾਬੀ', flag: '🇮🇳'}, {code: 'gu', name: 'ગુજરાતી', flag: '🇮🇳'}, {code: 'mr', name: 'मराठी', flag: '🇮🇳'} ]; // Template usage in bot response function botResponse(message, templateType = 'chat') { switch(templateType) { case 'question': return `<div class="bot-question"> <div class="question-text">${message}</div> <div class="question-options"></div> </div>`; case 'summary': return `<div class="summary-card"> <h3>Summary</h3> <div class="summary-content">${message}</div> </div>`; default: return `<div class="bot-message">${message}</div>`; } }
