// Template Generator Function function generateTemplate(type, data) { const templates = { 'language': ` <div class="language-selector"> <h3>${data.title || 'Select Language'}</h3> <div class="languages"> ${data.languages.map(lang => ` <button class="lang-btn" data-lang="${lang.code}"> ${lang.flag} ${lang.name} </button> `).join('')} </div> </div> `, 'question': ` <div class="question-step" data-id="${data.id}"> <div class="progress">${data.current}/${data.total}</div> <h4>${data.question}</h4> <div class="options"> ${data.options.map(opt => ` <button class="option" data-value="${opt.value}"> ${opt.text} </button> `).join('')} </div> </div> `, 'summary': ` <div class="summary"> <h3>${data.title}</h3> <div class="details"> ${Object.entries(data.details).map(([key, value]) => ` <div class="detail-item"> <strong>${key}:</strong> ${value} </div> `).join('')} </div> </div> ` }; return templates[type] || ''; } // Example Usage const languageData = { title: 'Choose Your Language', languages: [ {code: 'hi', name: 'हिंदी', flag: '🇮🇳'}, {code: 'en', name: 'English', flag: '🇺🇸'}, {code: 'ta', name: 'தமிழ்', flag: '🇮🇳'} ] }; const questionData = { id: 'website_type', current: 1, total: 5, question: 'What type of website do you need?', options: [ {value: 'basic', text: 'Basic Static Website'}, {value: 'dynamic', text: 'Dynamic Website'}, {value: '3d', text: '3D Interactive Website'} ] }; // Generate and insert document.getElementById('container').innerHTML = generateTemplate('language', languageData) + generateTemplate('question', questionData);
