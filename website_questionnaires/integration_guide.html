// Template for bot messages const botTemplates = { greeting: `<div class="bot-message"> <div class="avatar">🤖</div> <div class="message-content"> <div class="text">{{message}}</div> <div class="time">{{time}}</div> </div> </div>`, question: `<div class="question-message" data-question-id="{{id}}"> <div class="avatar">❓</div> <div class="message-content"> <div class="text">{{question}}</div> <div class="options"> {{#options}} <button class="option-btn" data-value="{{value}}">{{text}}</button> {{/options}} </div> </div> </div>`, summary: `<div class="summary-message"> <div class="avatar">📊</div> <div class="message-content"> <h4>{{title}}</h4> <div class="summary-content"> {{#items}} <div class="summary-item"> <strong>{{label}}:</strong> {{value}} </div> {{/items}} </div> <button class="download-btn">Download PDF</button> </div> </div>` }; // Function to render template function renderTemplate(templateName, data) { const template = botTemplates[templateName]; const html = Mustache.render(template, data); return html; } // Add message to chat function addMessage(templateName, data, isUser = false) { const chatContainer = document.getElementById('chat-container'); const html = renderTemplate(templateName, data); chatContainer.innerHTML += html; // Scroll to bottom chatContainer.scrollTop = chatContainer.scrollHeight; // Add animation gsap.from(chatContainer.lastElementChild, { opacity: 0, y: 20, duration: 0.5 }); } // Example usage addMessage('greeting', { message: 'Hello! How can I help you today?', time: new Date().toLocaleTimeString() }); addMessage('question', { id: 'website_type', question: 'What type of website do you need?', options: [ { value: 'basic', text: 'Basic Static Website' }, { value: 'dynamic', text: 'Dynamic Website' }, { value: '3d', text: '3D Interactive Website' } ] });

🎯 Questionnaire Integration

Integrate the questionnaire system:

// Questionnaire Configuration const questionnaires = { basic: { id: 'basic', name: 'Basic Static Website', questions: [ { id: 'business_name', text: 'What is your business name?', type: 'text', required: true }, { id: 'website_pages', text: 'What pages do you need?', type: 'checkbox', options: ['Home', 'About', 'Services', 'Portfolio', 'Contact', 'Blog', 'Testimonials'] }, { id: 'budget_range', text: 'What is your budget range?', type: 'select', options: ['₹10,000-₹25,000', '₹25,000-₹50,000', '₹50,000+'] } ] }, dynamic: { id: 'dynamic', name: 'Dynamic Website', questions: [ { id: 'features', text: 'What features do you need?', type: 'checkbox', options: ['User Login', 'CMS', 'E-commerce', 'Database', 'API Integration'] }, { id: 'budget_range', text: 'What is your budget range?', type: 'select', options: ['₹50,000-₹1,00,000', '₹1,00,000-₹2,50,000', '₹2,50,000-₹5,00,000'] } ] }, '3d': { id: '3d', name: '3D Interactive Website', questions: [ { id: '3d_style', text: 'What 3D style do you prefer?', type: 'select', options: ['Realistic', 'Stylized', 'Low-poly', 'Abstract'] }, { id: 'budget_range', text: 'What is your budget range?', type: 'select', options: ['₹1,00,000-₹3,00,000', '₹3,00,000-₹6,00,000', '₹6,00,000-₹10,00,000'] } ] } }; // Current questionnaire state let currentQuestionnaire = null; let currentQuestionIndex = 0; let answers = {}; // Start questionnaire function startQuestionnaire(type) { currentQuestionnaire = questionnaires[type]; currentQuestionIndex = 0; answers = {}; showQuestion(currentQuestionIndex); } // Show question function showQuestion(index) { const question = currentQuestionnaire.questions[index]; const data = { question: question.text, current: index + 1, total: currentQuestionnaire.questions.length, options: question.options ? question.options.map(opt => ({ value: opt.toLowerCase().replace(/[^a-z0-9]/g, '_'), text: opt })) : [] }; addMessage('question', data); } // Save answer function saveAnswer(questionId, answer) { answers[questionId] = answer; // Move to next question or finish currentQuestionIndex++; if (currentQuestionIndex < currentQuestionnaire.questions.length) { showQuestion(currentQuestionIndex); } else { finishQuestionnaire(); } } // Finish questionnaire function finishQuestionnaire() { const summaryData = { title: 'Questionnaire Complete!', items: Object.entries(answers).map(([key, value]) => ({ label: key.replace(/_/g, ' '), value: value })) }; addMessage('summary', summaryData); } // Event listeners document.addEventListener('click', function(e) { if (e.target.classList.contains('option-btn')) { const questionId = e.target.closest('.question-message').dataset.questionId; const value = e.target.dataset.value; saveAnswer(questionId, value); } if (e.target.classList.contains('lang-btn')) { const langCode = e.target.dataset.lang; setLanguage(langCode); } });

🚀 Deployment

Deploy your integrated bot:

// Deployment Script (deploy.js) const fs = require('fs'); const path = require('path'); // Copy template files to build directory function copyTemplates() { const templatesDir = path.join(__dirname, 'templates'); const buildDir = path.join(__dirname, 'build'); if (!fs.existsSync(buildDir)) { fs.mkdirSync(buildDir, { recursive: true }); } // Copy all template files const templateFiles = fs.readdirSync(templatesDir); templateFiles.forEach(file => { const source = path.join(templatesDir, file); const dest = path.join(buildDir, file); fs.copyFileSync(source, dest); console.log(`Copied: ${file}`); }); console.log('✅ All templates copied to build directory'); } // Minify CSS and JS function minifyAssets() { // Add your minification logic here console.log('✅ Assets minified'); } // Create production build function buildProduction() { console.log('🚀 Building production version...'); copyTemplates(); minifyAssets(); console.log('✅ Production build complete!'); console.log('📁 Build directory: ./build'); } // Run build buildProduction(); // Deployment commands: // 1. npm run build // 2. vercel --prod (for Vercel) // 3. firebase deploy (for Firebase) // 4. aws s3 sync ./build s3://your-bucket (for AWS S3)

📞 Support & Troubleshooting

Common Issues & Solutions:

Issue: Templates not loading

Solution: Check CDN links and network connectivity

Issue: Language not switching

Solution: Clear localStorage and check translation files

Issue: 3D models not displaying

Solution: Check WebGL support and model file paths

Issue: Mobile responsiveness broken

Solution: Check viewport meta tag and CSS media queries
